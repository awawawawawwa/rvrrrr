2. Create `src/bridge/binary-resolver.ts`:
   ```typescript
   import { existsSync } from 'node:fs';
   import { join, dirname } from 'node:path';
   import { execSync } from 'node:child_process';
   import { fileURLToPath } from 'node:url';

   export function resolveBinaryPath(explicitPath?: string): string {
       // 1. Explicit path
       if (explicitPath) {
           if (!existsSync(explicitPath)) {
               throw new Error(`Renderer binary not found at explicit path: ${explicitPath}`);
           }
           return explicitPath;
       }

       // 2. Relative to package (development: cargo build output)
       const binaryName = process.platform === 'win32' ? 'tui-engine-renderer.exe' : 'tui-engine-renderer';
       const packageDir = dirname(fileURLToPath(import.meta.url));
       const devPaths = [
           join(packageDir, '..', '..', 'crates', 'renderer', 'target', 'debug', binaryName),
           join(packageDir, '..', '..', 'crates', 'renderer', 'target', 'release', binaryName),
       ];
       for (const p of devPaths) {
           if (existsSync(p)) return p;
       }

       // 3. PATH lookup
       try {
           const which = process.platform === 'win32' ? 'where' : 'which';
           const result = execSync(`${which} tui-engine-renderer`, { encoding: 'utf8' }).trim();
           if (result) return result.split('\n')[0]!;
       } catch { /* not in PATH */ }

       throw new Error(
           'Could not find tui-engine-renderer binary. ' +
           'Run `cargo build` in crates/renderer/ or pass binaryPath in options.'
       );
   }
   ```

1. Create `src/bridge/ipc-child.ts` implementing the `IpcRendererBridge` class:

   **Constructor / spawn:**
   - Accept `BridgeOptions`. Call `resolveBinaryPath(options.binaryPath)`.
   - Spawn with `child_process.spawn(binaryPath, [], { stdio: ['pipe', 'pipe', 'inherit'] })`.
   - Set `state = 'starting'`.
   - Set up stdout NDJSON reader (see below).
   - Set up `child.on('exit', (code, signal) => ...)` handler:
     - If state is not 'stopping', this is a crash: set `state = 'crashed'`, call `options.onError?.(new Error(...))`.
     - If state is 'stopping', set `state = 'stopped'`.
   - Set up `child.on('error', (err) => ...)` for spawn failures.
   - Return a Promise that resolves when `{type: "ready"}` is received (or rejects on timeout/crash).

   **NDJSON reader on child.stdout:**
   - `child.stdout!.setEncoding('utf8')`.
   - Accumulate data chunks and split by `\n`. Maintain a partial-line buffer for incomplete lines.
   - For each complete line, `JSON.parse` it as `RendererMessageOut`.
   - Route by `msg.type`:
     - `'ready'`: resolve the startup promise, set `state = 'ready'`.
     - `'rendered'`: update `lastAckedFrameId = msg.frameId`.
     - `'input'`: call `options.onInput?.(msg.event)`.
     - `'error'`: call `options.onError?.(new Error(msg.message))`.
     - `'fatal'`: call `options.onError?.(new Error(msg.message))`, expect child exit soon.

   **enqueueRender(root: WidgetNode):**
   - If state is not 'ready' or 'running', return (drop the frame).
   - Set `state = 'running'` if it was 'ready'.
   - Increment `nextFrameId`.
   - Store as `pendingFrame = { type: 'render', frameId: nextFrameId, root }`.
   - This overwrites any previous pending frame (single-slot, no queue).

   **Send loop (throttling, BRDG-04):**
   - Start a `setInterval` at `1000 / maxFps` ms (default ~33ms for 30fps).
   - On each tick:
     - If `pendingFrame` is null, skip.
     - If `child.stdin` is not writable, skip.
     - Check backpressure: only send if `lastAckedFrameId >= lastSentFrameId` OR if `safetyTimeoutElapsed` (100ms since last send without ack).
     - Write `JSON.stringify(pendingFrame) + '\n'` to `child.stdin`.
     - Check return value of `.write()` — if `false`, listen for `'drain'` before next write (set a `drainPending` flag).
     - Set `lastSentFrameId = pendingFrame.frameId`.
     - Clear `pendingFrame = null`.

   **shutdown() / destroy():**
   - If already stopping/stopped, return.
   - Set `state = 'stopping'`.
   - Clear the send interval.
   - Remove signal handlers (important: don't leak).
   - Write `{ type: 'shutdown' }` as NDJSON to `child.stdin`.
   - Then `child.stdin.end()` to signal EOF.
   - Set a kill timeout (2 seconds): if child hasn't exited, `child.kill('SIGKILL')`.
   - Return a Promise that resolves when child exits.

   **Signal handlers (BRDG-03):**
   - On construction, register handlers for `SIGINT`, `SIGTERM`, and `uncaughtException` via `process.on(...)`.
   - Each handler calls `this.shutdown()`.
   - For `uncaughtException`: call shutdown, then re-throw (or `process.exit(1)`) after cleanup.
   - Store handler references so they can be removed in `shutdown()` to avoid leaking listeners.
   - On Windows, `SIGINT` works with `process.on('SIGINT', ...)`. `SIGTERM` may not fire on Windows — that's acceptable for v1.

   **sendResize(width: number, height: number):**
   - Write `{ type: 'resize', width, height }` NDJSON to child.stdin immediately (not throttled).
